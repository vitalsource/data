response_matrix = pd.DataFrame( index=data_set.student.unique(), columns=range( len( question_type ) ) )
for row in data_set.itertuples():
    student = row.student
    question = row.page_question_number
    score = row.correct
    response_matrix.loc[ student, question ] = score
response_matrix.index.name = 'student'
